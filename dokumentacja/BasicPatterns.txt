Sequence(f1,f2)={f1,f2,[](f1=><>f2),[](~f1=>~<>f2),[]~(f1 & f2)}

ParallelSplit(f1,f2,f3)={f1,f2  |  f3, [](f1=><>f2  &  <> f3),
[](~f1=>~<>f2 & ~<>f3),[]~(f1 & (f2  |  f3))}

Synchronization(f1,f2,f3)={f1  |  f2, f3,[](f1  & f2=><>f3),
[](~(f1 & f2)=>~<>f3,[]~((f1 | f2) & f3)}



SimpleMerge(f1,f2,f3)={f1  |  f2,f3,[](f1  |  f2=><>f3),
[](~(f1 | f2)=>~<>f3),[]~(f1 | f2),[]~()(f1 | f2) & f3 }
